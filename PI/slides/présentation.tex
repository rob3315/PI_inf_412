 \begin{frame}{DSATUR}{Introduction}
      \begin{block}{Main Idea}<1->
        Greedy algorithm :\\
        the harder it is to color a vertex, the quicker it should be done
      \end{block}

      \begin{block}{DSATUR}<2->
        the more colored neighbors a vertex has, the harder it is to color.
      \end{block}
\end{frame}
 \begin{frame}{Peudo-Code}{Introduction}
 \begin{algorithm}[H]
    \caption{DSATUR}
    \begin{algorithmic}
        \Procedure{DSATUR}{}\newline
    \While{some vertices are uncolored}
        \State find the vertex with the most colored neighbors \Comment{$O(n)$}
        \State visit the neighbors and store the unavailable colors\Comment{$O(D \log D)$}
        \If{a hint can be satisfied} \Comment{$O(\textcolor{red}{H* \log D})$}   		\State do it	\Else   		\State find the smallest available color \Comment{$O(D \log D)$}
        \EndIf
        \State update the comparator \Comment{$O(D)$}
        \EndWhile
        \EndProcedure
    \end{algorithmic}
    \label{alg_1}
\end{algorithm}
 \end{frame}
